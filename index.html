    <script>

        // Sample Data: Replace with actual values

        const items = [

            { name: "Full Panels", price: 348 },

            { name: "Classic Collection", price: 0.1 },

        ];

        const tiers = { 

            "Full Retail": 1.0, 

            "40% OFF": 0.6, 

            "Plus 15% OFF": 0.85, 

            "Plus 10% OFF": 0.9, 

            "Plus 5% OFF": 0.95 

        };


        // Populate Items Table

        const itemTableBody = document.getElementById("itemTable").querySelector("tbody");

        items.forEach((item, index) => {

            const row = document.createElement("tr");

            row.innerHTML = `

                <td><input type="checkbox" data-index="${index}"></td>

                <td>${item.name}</td>

                <td><input type="number" value="0" min="0" data-index="${index}" class="quantity"></td>

                <td>$${item.price.toFixed(2)}</td>

            `;

            itemTableBody.appendChild(row);

        });


        // Calculate Totals

        function calculateTotals() {

            const quantities = document.querySelectorAll(".quantity");

            let totals = { "Full Retail": 0, "40% OFF": 0, "Plus 15% OFF": 0, "Plus 10% OFF": 0, "Plus 5% OFF": 0 };


            quantities.forEach((input) => {

                const index = input.dataset.index;

                const quantity = parseFloat(input.value) || 0;

                const price = items[index].price;


                if (document.querySelector(`input[type="checkbox"][data-index="${index}"]`).checked) {

                    Object.keys(tiers).forEach((tier) => {

                        totals[tier] += quantity * price * tiers[tier];

                    });

                }

            });


            // Update Totals Table

            document.getElementById("fullRetailTotal").innerText = `$${totals["Full Retail"].toFixed(2)}`;

            document.getElementById("40OffTotal").innerText = `$${totals["40% OFF"].toFixed(2)}`;

            document.getElementById("15OffTotal").innerText = `$${totals["Plus 15% OFF"].toFixed(2)}`;

            document.getElementById("10OffTotal").innerText = `$${totals["Plus 10% OFF"].toFixed(2)}`;

            document.getElementById("5OffTotal").innerText = `$${totals["Plus 5% OFF"].toFixed(2)}`;

        }
